import java.util.Scanner; 
public class Tabuada {
	public static void main (String[] args) {

	Scanner in = new Scanner (System.in); 
	final int TAM = 5;
	int i, A[], t, n; 
	
	A = new int [TAM];

	System.out.println ("Digite 5 n√∫meros inteiros: ");
	
	for (i=0; i<TAM; i++){
		A[i] = in.nextInt();
	} 

	for (i=0; i<TAM; i++) {
	System.out.println ("Tabuada do " + A);
	}

	for (n=0; n<=10; n++) {
	t = A[i] * n;
	System.out.println (A[i] + "x" + n + "=" + t);
	}

	}
}
